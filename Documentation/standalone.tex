% ==================
% Standalone support
% ==================


\makeatletter
\ifdefined\preloaded
\else
	\newcommand{\preloaded}{not set!}
	\ifx\documentclass\@twoclasseserror % after \documentclass
		\renewcommand{\preloaded}{true}
	\else
		\input{../preamble}
  		\begin{document}
		\renewcommand{\preloaded}{false}
	\fi
\fi
\makeatother
\trysetmain{}
